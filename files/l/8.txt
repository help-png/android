MainActivity.java
package com.example.twofragmentcolorapp;

import androidx.appcompat.app.AppCompatActivity;
import android.graphics.Color;
import android.os.Bundle;
import android.widget.FrameLayout;
import android.widget.TextView;

public class MainActivity extends AppCompatActivity
        implements BgFragment.BackgroundListener, TextFragment.TextColorListener {

    FrameLayout root;
    TextView sampleText;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        root = findViewById(R.id.rootLayout);
        sampleText = findViewById(R.id.sampleText);

        if(savedInstanceState == null){
            getSupportFragmentManager().beginTransaction()
                    .replace(R.id.bgContainer, new BgFragment())
                    .replace(R.id.textContainer, new TextFragment())
                    .commit();
        }
    }

    @Override
    public void changeBackground(int color) {
        root.setBackgroundColor(color);
    }

    @Override
    public void changeTextColor(int color) {
        sampleText.setTextColor(color);
    }
}

✔ BgFragment.java (Fragment to set Background)
package com.example.twofragmentcolorapp;

import android.content.Context;
import android.graphics.Color;
import android.os.Bundle;
import androidx.fragment.app.Fragment;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.Button;

public class BgFragment extends Fragment {

    BackgroundListener listener;

    public interface BackgroundListener {
        void changeBackground(int color);
    }

    @Override
    public void onAttach(Context context) {
        super.onAttach(context);
        listener = (BackgroundListener) context;
    }

    @Override
    public View onCreateView(LayoutInflater inflater, ViewGroup container,
                             Bundle savedInstanceState) {
        View v = inflater.inflate(R.layout.fragment_bg, container, false);

        Button b1 = v.findViewById(R.id.btnWhite);
        Button b2 = v.findViewById(R.id.btnYellow);
        Button b3 = v.findViewById(R.id.btnGray);

        b1.setOnClickListener(view -> listener.changeBackground(Color.WHITE));
        b2.setOnClickListener(view -> listener.changeBackground(Color.YELLOW));
        b3.setOnClickListener(view -> listener.changeBackground(Color.LTGRAY));

        return v;
    }
}

✔ TextFragment.java (Fragment to set Text Foreground Color)
package com.example.twofragmentcolorapp;

import android.content.Context;
import android.graphics.Color;
import android.os.Bundle;
import androidx.fragment.app.Fragment;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.Button;

public class TextFragment extends Fragment {

    TextColorListener listener;

    public interface TextColorListener {
        void changeTextColor(int color);
    }

    @Override
    public void onAttach(Context context) {
        super.onAttach(context);
        listener = (TextColorListener) context;
    }

    @Override
    public View onCreateView(LayoutInflater inflater, ViewGroup container,
                             Bundle savedInstanceState) {
        View v = inflater.inflate(R.layout.fragment_text, container, false);

        Button b1 = v.findViewById(R.id.btnBlack);
        Button b2 = v.findViewById(R.id.btnRed);
        Button b3 = v.findViewById(R.id.btnBlue);

        b1.setOnClickListener(view -> listener.changeTextColor(Color.BLACK));
        b2.setOnClickListener(view -> listener.changeTextColor(Color.RED));
        b3.setOnClickListener(view -> listener.changeTextColor(Color.BLUE));

        return v;
    }
}
