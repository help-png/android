import java.applet.Applet;
import java.awt.*;
import java.awt.event.*;

/* <applet code="AppletFrameDemo" width=400 height=300> </applet> */

public class AppletFrameDemo extends Applet implements MouseListener {

    Frame f;

    @Override
    public void init() {
        addMouseListener(this);

        f = new Frame("Child Frame");
        f.setSize(300, 200);
        f.setLayout(new FlowLayout());
        f.add(new Label("Mouse is inside the Applet"));
        f.setVisible(false);   // hidden initially
    }

    @Override
    public void mouseEntered(MouseEvent e) {
        f.setVisible(true);   // show frame
    }

    @Override
    public void mouseExited(MouseEvent e) {
        f.setVisible(false);  // hide frame
    }

    @Override
    public void mouseClicked(MouseEvent e) {
        showStatus("Mouse Clicked inside Applet");
    }

    @Override
    public void mousePressed(MouseEvent e) { }

    @Override
    public void mouseReleased(MouseEvent e) { }
}
